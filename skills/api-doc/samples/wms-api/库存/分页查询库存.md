## 请求参数
```yaml
type: object
properties:
  apiName:
    type: string
    description: 请求接口名称
    default: hy_wms_page_query_inventories_WebApi
  outType:
    type: string
    description: 请求接口返回格式
    default: json
  data:
    type: object
    description: 请求数据对象
    properties:
      page:
        type: integer
        description: 页数
        default: 1
      pageSize:
        type: integer
        description: 每页数量
        default: 10
      conditions:
        type: array
        description: 查询条件组
        items:
          type: object
          properties:
            factoryCode:
              type: string
              description: 工厂编码
            itemTypeCode:
              type: string
              description: 物料类型编码
            warehouseCode:
              type: string
              description: 仓库编码
            materialCode:
              type: string
              description: 物料编码
            lot:
              type: string
              description: 批次号
            sn:
              type: string
              description: 序列号
required: [apiName, outType, data]
```

## 请求示例
```json
{
  "apiName": "hy_wms_page_query_inventories_WebApi",
  "outType": "json",
  "data": {
    "page": 1,
    "pageSize": 10,
    "conditions": [
      {
        "factoryCode": "F001",
        "warehouseCode": "WH01",
        "materialCode": "M001"
      }
    ]
  }
}
```

## 响应参数
```yaml

  code:
    type: integer
    description: 状态码
  message:

  data:
    type: object
    description: 数据对象
    properties:
      code:
        type: string
        description: 返回状态 (200:成功, 400:失败)
      message:
        type: string
        description: 返回提示信息
      data:
        type: object
        description: 分页数据
        properties:
          total:
            type: integer
            description: 总数量
          page:

            description: 当前页数
          pageSize:
            type: integer
            description: 每页数量
          data:
            type: array
            description: 库存列表
            items:
              type: object
              properties:
                warehouseCode:
                  type: string
                  description: 仓库编码
                warehouseName:
                  type: string
                  description: 仓库名称
                materialCode:
                  type: string
                  description: 物料编码
                materialName:
                  type: string
                  description: 物料名称
                specification:
                  type: string
                  description: 规格型号
```
