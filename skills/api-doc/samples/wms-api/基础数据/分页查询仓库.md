## 请求参数
```yaml
type: object
properties:
  apiName:
    type: string
    description: 请求接口名称
    default: hy_wms_page_query_warehouses_WebApi
  outType:
    type: string
    description: 请求接口返回格式
    default: json
  data:
    type: object
    description: 请求数据对象
    properties:
      page:
        type: integer
        description: 页数
        default: 1
      pageSize:
        type: integer
        description: 每页数量
        default: 10
        maximum: 100
      conditions:
        type: array
        description: 查询条件组
        items:
          type: object
          properties:
            factoryCode:
              type: string
              description: 工厂编码
            itemTypeCode:
              type: string
              description: 物料类型编码
            code:
              type: string
              description: 仓库编码
required: [apiName, outType, data]
```

## 请求示例
```json
{
  "apiName": "hy_wms_page_query_warehouses_WebApi",
  "outType": "json",
  "data": {
    "page": 1,
    "pageSize": 10,
    "conditions": [
      {
        "factoryCode": "F001",
        "itemTypeCode": "RAW",
        "code": "WH01"
      }
    ]
  }
}
```

## 响应参数
```yaml
type: object
properties:
  code:
    type: integer
    description: 状态码
  message:
    type: string
    description: 消息
  data:
    type: object
    description: 数据对象
    properties:
      code:
        type: string
        description: 返回状态 (200:成功, 400:失败)
      message:
        type: string
        description: 返回提示信息
      data:
        type: object
        description: 分页数据
        properties:
          total:
            type: integer
            description: 总数量
          page:
            type: integer
            description: 当前页数
          pageSize:
            type: integer
            description: 每页数量
          data:
            type: array
            description: 仓库列表
            items:
              type: object
              properties:
                code:
                  type: string
                  description: 仓库编码
                no:
                  type: string
                  description: 仓库码
                name:
                  type: string
                  description: 仓库名称
                itemTypeCode:
                  type: string
                  description: 物料类型编码
                itemTypeName:
                  type: string
                  description: 物料类型名称
                factoryCode:
                  type: string
                  description: 工厂编码
```
